#!/bin/bash

# Read the input number from the user
read -p "Enter a number: " num

# Initialize a flag variable to indicate whether the number is prime or not
is_prime=1

# Check if the number is divisible by any number other than 1 and itself
for (( i=2; i<num; i++ )); do
  if (( num % i == 0 )); then
    is_prime=0
    break
  fi
done

# Display the result
if (( is_prime == 1 )); then
  echo "$num is a prime number"
else
  echo "$num is not a prime number"
fi

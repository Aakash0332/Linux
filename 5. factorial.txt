#!/bin/bash

# Read the input number from the user
read -p "Enter a number: " num

# Initialize the factorial to 1
fact=1

# Loop over the numbers from 1 to the input number, and multiply the factorial by each number
for (( i=1; i<=num; i++ )); do
  fact=$(( fact * i ))
done

# Display the result
echo "The factorial of $num is: $fact"
